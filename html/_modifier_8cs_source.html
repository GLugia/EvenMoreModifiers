151  protected internal static Modifier _Load(Item item, TagCompound tag)
152  {
153  string modName = tag.GetString("ModName");
154 
155  if (RegistryLoader.Mods.TryGetValue(modName, out var assembly))
156  {
157  // If we load a null here, it means a modifier is unloaded
158  Modifier m = null;
159 
160  var saveVersion = tag.ContainsKey("ModifierSaveVersion") ? tag.GetInt("ModifierSaveVersion") : 1;
161 
162  string modifierTypeName = tag.GetString("Type");
163 
164  // adapt by save version
165  if (saveVersion == 1)
166  {
167  // in first save version, modifiers were saved by full assembly namespace
168  //m = (ModifierPool)Activator.CreateInstance(assembly.GetType(tag.GetString("Type")));// we modified saving
169  modifierTypeName = modifierTypeName.Substring(modifierTypeName.LastIndexOf('.') + 1);
170  m = ContentLoader.Modifier.GetContent(modName, modifierTypeName);
171  }
172  else if (saveVersion >= 2)
173  {
174  // from saveVersion 2 and onwards, they are saved by assembly (mod) and type name
175  m = ContentLoader.Modifier.GetContent(modName, modifierTypeName);
176  }
177 
178  if (m != null)
179  {
180  // saveVersion 1, no longer needed. Type and Mod is already created by new instance
181  //m.Type = tag.Get<uint>("ModifierType");
182  //m.Mod = ModLoader.GetMod(modname);
183  var p = ModifierProperties._Load(item, tag.GetCompound("ModifierProperties"));
184  m.Properties = m.GetModifierProperties(item).Build();
185  m.Properties.Magnitude = p.Magnitude;
186  m.Properties.Power = p.Power;
187  m.Load(item, tag);
188  return m;
189  }
190 
191  return null;
192  }
193 
194  Loot.Logger.ErrorFormat("There was a load error for modifier, TC: {0}", tag);
195  return null;
