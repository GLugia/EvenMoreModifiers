47  internal ModifierPool RollNewPool(ModifierContext ctx, ItemRollProperties itemRollProperties = null)
48  {
49  // Default to normal
50  if (itemRollProperties == null)
51  {
52  itemRollProperties = new ItemRollProperties();
53  }
54 
55  // Now we have actually rolled
56  HasRolled = true;

58  // Rolling logic
59  bool noForce = true;
60 
61  // Custom roll behavior provided
62  if (itemRollProperties.OverrideRollModifierPool != null)
63  {
64  ModifierPool = itemRollProperties.OverrideRollModifierPool.Invoke();
65  noForce = !ModifierPool?._CanRoll(ctx) ?? true;
66  }
67 
68  // No behavior provided
69  if (noForce)
70  {
71  // A pool is forced to roll
72  if (itemRollProperties.ForceModifierPool != null)
73  {
74  ModifierPool = ContentLoader.ModifierPool.GetContent(itemRollProperties.ForceModifierPool.GetType());
75  noForce = !ModifierPool?._CanRoll(ctx) ?? true;
76  }
77 
78  // No pool forced to roll or it's not valid
79  if (noForce)
80  {
81  // Try rolling a predefined (weighted) pool
82  bool rollPredefinedPool = Main.rand.NextFloat() <= itemRollProperties.RollPredefinedPoolChance;
83  noForce = !rollPredefinedPool;
84 
85  if (rollPredefinedPool)
86  {
87  // GetWeightedPool already checks _CanRoll
88  ModifierPool = ContentLoader.ModifierPool.GetWeightedPool(ctx);
89  noForce = ModifierPool == null || !ModifierPool._CanRoll(ctx);
90  }
91 
92  // Roll from all modifiers
93  if (noForce)

95  ModifierPool = Loot.Instance.GetModifierPool<AllModifiersPool>();
96  if (!ModifierPool._CanRoll(ctx))
97  {
98  ModifierPool = null;
99  return null;
100  }
101  }
102  }
103  }
104

106  if (!RollNewModifiers(ctx, itemRollProperties))
107  {

109  }
110  else
111  {
112  ModifierPool.UpdateRarity();
113  }

116  ctx.Item.GetGlobalItem<GlowmaskGlobalItem>().NeedsUpdate = true;

127  internal bool RollNewModifiers(ModifierContext ctx, ItemRollProperties itemRollProperties)
128  {
129  // Firstly, prepare a WeightedRandom list with modifiers
130  // that are rollable in this context
131  WeightedRandom<Modifier> wr = new WeightedRandom<Modifier>();
132  List<Modifier> list = new List<Modifier>();
133 
134  foreach (var e in ModifierPool.GetRollableModifiers(ctx))
135  {
136  wr.Add(e, e.Properties.RollChance);
137  }
138 
139  // Up to n times, try rolling a mod
140  // @ todo since we can increase lines rolled, make it so that MaxRollableLines influences the number of rows drawn in the UI
141  for (int i = 0; i < itemRollProperties.MaxRollableLines; ++i)
142  {
143  // If there are no mods left, or we fail the roll, break.
144  if (wr.elements.Count <= 0
145  || !_forceNextRoll
146  && i > 0
147  && list.Count >= itemRollProperties.MinModifierRolls
148  && Main.rand.NextFloat() > itemRollProperties.RollNextChance)
149  {
150  break;
151  }
152 
153  _forceNextRoll = false;
154 
155  // Get a next weighted random mod
156  // Clone the mod (new instance) and roll it's properties, then roll it
157  Modifier e = wr.Get();
158  Modifier eClone = (Modifier)e.Clone();
159  float luck = itemRollProperties.ExtraLuck;
160  if (ctx.Player != null)
161  {
162  luck += ModifierPlayer.Player(ctx.Player).GetEffect<LuckEffect>().Luck;
163  }
164  eClone.Properties =
165  eClone.GetModifierProperties(ctx.Item).Build()
166  .RollMagnitudeAndPower(
167  magnitudePower: itemRollProperties.MagnitudePower,
168  lukStat: luck);
169  eClone.Roll(ctx, list);
170 
171  // If the mod deemed to be unable to be added,
172  // Force that the next roll is successful
173  // (no RNG on top of RNG)
174  if (!eClone.PostRoll(ctx, list))
175  {
176  _forceNextRoll = true;
177  continue;
178  }
179 
180  // The mod can be added
181  list.Add(eClone);
182 
183  // If it is a unique modifier, remove it from the list to be rolled
184  if (eClone.Properties.IsUnique)
185  {
186  wr.elements.RemoveAll(x => x.Item1.Type == eClone.Type);
187  wr.needsRefresh = true;
188  }
189  }
190 
191  ModifierPool.ActiveModifiers = list.ToArray();
192  return list.Any();
193  }
