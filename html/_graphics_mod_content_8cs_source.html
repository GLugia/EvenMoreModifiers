244  //to clarify, this code simply 'fixes' out bad saving of the file in photoshop or whatever
245  //yeah, it works for every pixel individually
246  //some art programs save the file with every pixel having an RGBA where RGB is bigger than 0 but A may be 0 , where A would usually be transparency there, in XNA's texture loading the alpha channel is independent , and with terraria's draw mode it actually can do awesome things
247  //in terraria, A channel signals additive<---> opaque
248  //a pixel with 255 alpha draws fully opaque, a pixel with 0 alpha draws fully additive
249  //so if you do say, 255,0,0,0 color, it would draw a fully additive red
250  //which can be awesome...except when your file saves the alpha channel of supposedly transparent pixels wrong
251  public static Texture2D MultiplyColorsByAlpha(Texture2D texture)
252  {
253  Color[] data = new Color[texture.Width * texture.Height];
254  texture.GetData(data);
255  for (int i = 0; i < data.Length; i++)
256  {
257  Vector4 we = data[i].ToVector4();
258  data[i] = new Color(we.X * we.W, we.Y * we.W, we.Z * we.W, we.W);
259  }
260  texture.SetData(data);
261  return texture;
262  }
