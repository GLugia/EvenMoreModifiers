149  protected internal static Modifier _Load(Item item, TagCompound tag)
150  {
151  string modName = tag.GetString("ModName");
152  if (modName != null
153  && MainLoader.Mods.TryGetValue(modName, out var assembly))
154  {
155  // If we load a null here, it means a modifier is unloaded
156  Modifier m = null;
157 
158  var saveVersion = tag.ContainsKey("ModifierSaveVersion") ? tag.GetInt("ModifierSaveVersion") : 1;
159 
160  string modifierTypeName = tag.GetString("Type");
161 
162  // adapt by save version
163  if (saveVersion == 1)
164  {
165  // in first save version, modifiers were saved by full assembly namespace
166  //m = (ModifierPool)Activator.CreateInstance(assembly.GetType(tag.GetString("Type")));// we modified saving
167  modifierTypeName = modifierTypeName.Substring(modifierTypeName.LastIndexOf('.') + 1);
168  m = ContentLoader.Modifier.GetContent(modName, modifierTypeName);
169  }
170  else if (saveVersion >= 2)
171  {
172  // from saveVersion 2 and onwards, they are saved by assembly (mod) and type name
173  m = ContentLoader.Modifier.GetContent(modName, modifierTypeName);
174  }
175 
176  if (m != null)
177  {
178  // saveVersion 1, no longer needed. Type and Mod is already created by new instance
179  //m.Type = tag.Get<uint>("ModifierType");
180  //m.Mod = ModLoader.GetMod(modname);
181  var p = ModifierProperties._Load(item, tag.GetCompound("ModifierProperties"));
182  m.Properties = m.GetModifierProperties(item).Build();
183  m.Properties.Magnitude = p.Magnitude;
184  m.Properties.Power = p.Power;
185  m.Load(item, tag);
186  return m;
187  }
188 
189  return null;
190  }
191 
192  Log4c.Logger.ErrorFormat("There was a load error for modifier, TC: {0}", tag);
193  return null;
