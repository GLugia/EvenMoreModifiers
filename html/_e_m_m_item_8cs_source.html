59  internal ModifierPool RollNewPool(ModifierContext ctx, ItemRollProperties itemRollProperties = null)

64  itemRollProperties = new ItemRollProperties();

77  noForce = !ModifierPool?._CanRoll(ctx) ?? true;
78  }
79

81  if (noForce)

87  noForce = !ModifierPool?._CanRoll(ctx) ?? true;

91  if (noForce)
92  {

94  bool rollPredefinedPool = Main.rand.NextFloat() <= itemRollProperties.RollPredefinedPoolChance;
95  noForce = !rollPredefinedPool;
96 
97  if (rollPredefinedPool)
98  {
99  // GetWeightedPool already checks _CanRoll
100  ModifierPool = ContentLoader.ModifierPool.GetWeightedPool(ctx);
101  noForce = ModifierPool == null || !ModifierPool._CanRoll(ctx);
102  }
103 
104  // Roll from all modifiers
105  if (noForce)
106  {

108  if (!ModifierPool._CanRoll(ctx))
109  {
110  ModifierPool = null;
111  return null;
112  }
113  }
114  }
115  }
116 
117  // Attempt rolling modifiers
118  if (!RollNewModifiers(ctx, itemRollProperties))
119  {
120  ModifierPool = null; // reset (didn't roll anything)
121  }
122  else
123  {
124  ModifierPool.UpdateRarity();
125  }
126

139  internal bool RollNewModifiers(ModifierContext ctx, ItemRollProperties itemRollProperties)
140  {
141  // Firstly, prepare a WeightedRandom list with modifiers
142  // that are rollable in this context
143  WeightedRandom<Modifier> wr = new WeightedRandom<Modifier>();
144  List<Modifier> list = new List<Modifier>();
145 
146  foreach (var e in ModifierPool.GetRollableModifiers(ctx))
147  {
148  wr.Add(e, e.Properties.RollChance);
149  }
150 
151  // Up to n times, try rolling a mod
152  // @ todo since we can increase lines rolled, make it so that MaxRollableLines influences the number of rows drawn in the UI
153  for (int i = 0; i < itemRollProperties.MaxRollableLines; ++i)
154  {
155  // If there are no mods left, or we fail the roll, break.
156  if (wr.elements.Count <= 0
157  || !_forceNextRoll
158  && i > 0
159  && list.Count >= itemRollProperties.MinModifierRolls
160  && Main.rand.NextFloat() > itemRollProperties.RollNextChance)
161  {
162  break;
163  }
164 
165  _forceNextRoll = false;
166 
167  // Get a next weighted random mod
168  // Clone the mod (new instance) and roll it's properties, then roll it
169  Modifier e = wr.Get();
170  Modifier eClone = (Modifier)e.Clone();
171  float luck = itemRollProperties.ExtraLuck;
172  if (ctx.Player != null)
173  {
174  luck += ModifierPlayer.Player(ctx.Player).GetEffect<LuckEffect>().Luck;
175  }
176  eClone.Properties =
177  eClone.GetModifierProperties(ctx.Item)
178  .RollMagnitudeAndPower(
179  magnitudePower: itemRollProperties.MagnitudePower,
180  lukStat: luck);
181  eClone.Roll(ctx, list);
182 
183  // If the mod deemed to be unable to be added,
184  // Force that the next roll is successful
185  // (no RNG on top of RNG)
186  if (!eClone.PostRoll(ctx, list))
187  {
188  _forceNextRoll = true;
189  continue;
190  }
191 
192  // The mod can be added
193  list.Add(eClone);
194 
195  // If it is a unique modifier, remove it from the list to be rolled
196  if (eClone.Properties.UniqueModifier)
197  {
198  wr.elements.RemoveAll(x => x.Item1.Type == eClone.Type);
199  wr.needsRefresh = true;
200  }
201  }
202 
203  ModifierPool.ActiveModifiers = list.ToArray();
204  return list.Any();
205  }
